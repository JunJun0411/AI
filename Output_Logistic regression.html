# X_train, y_train, X_test, y_test, epoch, learningRate, Single True, Snum: Single class number
def Logistic(X_train, y_train, X_test, y_test, epoch, learningRate, Single, Snum):
    #Single Class  Class  inputdata Slicing
    if Single == True:
        y_train = np.eye(np.unique(y_train).shape[0], dtype = int)[y_train]
        y_test = np.eye(np.unique(y_test).shape[0], dtype = int)[y_test]
        y_train = y_train[:, Snum]
        y_test = y_test[:, Snum]
    # Logistic Regression    
    lo = Logistic_Regression(X_train, y_train, learningRate, Single)
    lo.learn(epoch) # epoch 
    if Single == True:
        print(Snum," Class, epoch = ", epoch, "LearningRate = ", learningRate)
    else:
        print("MultiClass, epoch = ", epoch, "LearningRate = ", learningRate)
    lo.predict(X_test, y_test)
    y = lo.cost2
    x = np.arange(epoch)
    plt.plot(x, y)
    plt.show()

import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D
from sklearn.datasets import load_iris

class Logistic_Regression():
    def __init__(self, X_train, y_train, learningRate, S):
        self.X_train = self.Xmake(X_train) # x0 
        self.y_train = self.one_hot(y_train) # , 0 1  matrix .
        self.W = np.random.randn(X_train.shape[1] + 1, np.unique(y_train).shape[0]) #  Weight ( # of feature by # of target )
        self.learningRate = learningRate
        self.h = []
        self.cost2=[]
        self.EMIN = -np.log(np.finfo(type(0.1)).max)
        self.Single = S
    
    # X_train  bias 
    def Xmake(self, X_train):
        # z = [ [1], ... , [1] ] m , shape = (m, 1)
        z = np.ones(X_train.shape[0]).reshape(-1,1) 
        return np.concatenate((z, X_train), axis=1)
    
    def one_hot(self, y_train):
        #      (2)
        if y_train.ndim == 2:
            return y_train
        
        # 1 ndarray 
        return np.eye(np.unique(y_train).shape[0])[y_train]
    
    def sigmoid(self, x):
        xSafe = np.array(np.maximum(x, self.EMIN))
        return(1.0/(1+np.exp(-xSafe)))

    def cost(self):
        # CostFunction, J()   
        vec1 = -(np.log(self.h) * self.y_train)
        J1 = vec1.sum(0) / self.y_train.sum(0)
        vec2 = -(np.log(np.maximum(1 - self.h, 1e-8)) * (1-self.y_train)) # log(0)  1e-8
        J2 = vec2.sum(0) / self.y_train.sum(0)
        #  Class   Class logh(x) +  Class 1-logh(x)
        J = J1 + J2
        if self.Single == True:
            J = J[0]
        self.cost2.append(J) # plt 
        return J
        
    def learn(self, epoch):
        for i in range(epoch):
            # sigmoid  h(w*x) => m*t matrix
            self.h = self.sigmoid(np.dot(self.X_train, self.W))
            self.W -= self.learningRate * np.dot(self.X_train.T,(self.h - self.y_train))
            self.cost()
        
    def predict(self, X_test, y_test):
        # MultiClass
        if self.Single == False:
            X = self.Xmake(X_test)
            # X_test weight 
            Xweight = np.dot(X, self.W)
            #    class 
            y_predict = np.argmax(Xweight, axis=1)
            # Accuracy %
            return np.sum((y_test - y_predict)==0) / y_test.shape[0] * 100
        
        else :
            X = self.Xmake(X_test)
            # X_test weight 
            Xweight = np.dot(X, self.W)
            # Xweight sigmoid    0.5  
            y_predict = np.array(self.sigmoid([Xweight])>=0.5)
            y_acc = np.equal(y_predict.reshape(-1, 2), self.one_hot(y_test))
            # Accuracy %
            return y_acc.sum(0)[0] / y_test.shape[0] * 100
            
    def flush(self):
        self.W = []
        self.h = []
